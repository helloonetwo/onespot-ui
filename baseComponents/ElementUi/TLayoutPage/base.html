        },

            valueFormat: 'yyyy-MM-dd'

      return {
        workshopId,
        stocktakeType,
        stocktakeJobStatus,
        warehouseCode,
        warehouseName,
        stocktakePlanNo,
        stocktakeStartTime: planDate && planDate[0] ? planDate[0] : null,
        stocktakeEndTime: planDate && planDate[1] ? planDate[1] : null,
        lookType: 1,
      }
    }

    this.getStockTakeList()

      }
      // console.log(3333, this.table)
      this.loading = false
